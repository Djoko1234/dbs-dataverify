function doLogin(){
  const u=document.getElementById('au').value.trim();
  const p=document.getElementById('ap').value;
  const ue=document.getElementById('ae').value;
  const found=USERS_DB.find(x=>x.username===u&&x.password===p&&x.active);
  if(!found){
    document.getElementById('auth-err').style.display='block';

    return;
  }
  APP.user=found;
  found.lastLogin=new Date().toISOString();
  document.getElementById('hav').textContent=found.name.split(' ').map(s=>s[0]).join('').toUpperCase().slice(0,2);
  document.getElementById('hname').textContent=found.name;
  document.getElementById('hrole').textContent=found.entity+' · '+found.role.charAt(0).toUpperCase()+found.role.slice(1);
  document.getElementById('t-test').value=found.name;
  document.getElementById('auth').style.display='none';
  document.getElementById('app').style.display='block';
  APP.queries=[...Q_INIT];
  APP.servers=[...SERVERS_INIT];
  APP.users=[...USERS_DB];
  APP.tests=[...TESTS_INIT];
  // Admin-only elements
  const isAdmin=found.role==='admin';
  document.querySelectorAll('.admin-only').forEach(el=>el.style.display=isAdmin?'':'none');
  sysLog('OK',`Connexion réussie — ${found.name} (${found.role}) depuis ${found.entity}`);
  initApp();
}

function goWiz(n){
  for(let i=1;i<=5;i++){
    const sc=document.getElementById('sc'+i);
    if(sc)sc.style.display=(i===n)?'block':'none';
    const ws=document.getElementById('ws'+i);

      barEl.classList.add('g');
      pctEl.textContent='100%';
      sysLog('OK',`Test ${ref} terminé — mode ${mode} — ${selectedQ.length} requêtes`);
      const newT={
        id:'T'+Date.now(),ref,filiale:fil,app,atDate,heure:atH,
        testeur:document.getElementById('t-test').value,
        validateur:document.getElementById('t-valid').value,

function addLog(el,type,msg){
  const d=document.createElement('div');

  el.appendChild(d);
  el.scrollTop=el.scrollHeight;
  sysLog(type,msg);
}

function execRestTest(id){
  const t=APP.pendingTests.find(x=>x.id===id);
  if(!t)return;
  const rh=document.getElementById('pr-host-'+id).value;
  const rs=document.getElementById('pr-sid-'+id).value;
  if(!rh){alert('Veuillez renseigner l\'hôte du serveur restauré.');return;}
  showLd('Exécution REST en cours…',`Connexion à ${rh} — at_date: ${t.atDate} ${t.heure}`);
  let p=0;
  const it=setInterval(()=>{
    p+=12;updLd(p);
    if(p>=100){
      clearInterval(it);hideLd();
      t.restHost=rh;t.restSid=rs;
      t.verdict=Math.random()>.2?'CONFORME':'PARTIEL';
      t.taux=+(94+Math.random()*5).toFixed(1);
      t.mode='both';
      APP.pendingTests=APP.pendingTests.filter(x=>x.id!==id);
      APP.lastComparison=t;
      updatePendBadge();
      renderDash();renderHist();renderPending();
      sysLog('OK',`Test REST exécuté pour ${t.ref} — verdict: ${t.verdict}`);
      nav('comp');buildCompTable(t);
      alert(`✅ Test REST terminé !\nVerdic : ${t.verdict} — ${t.taux}% de conformité`);
    }
  },200);
}

function exportSynthese(){
  const wb=XLSX.utils.book_new();
  const filialesAll=['GABON','MALI','CI','MADA','CAMEROUN','COMORES','GUINEE'];
  const ws=XLSX.utils.aoa_to_sheet([
    ['SYNTHÈSE MULTI-FILIALES — AFG BANK — DBS DATAVERIFY'],[''],
    ['Filiale','Tests','Conformes','Écarts','Taux moy','Dernière date'],
    ...filialesAll.map(f=>{
      const ts=APP.tests.filter(t=>t.filiale===f);
      const ok=ts.filter(t=>t.verdict==='CONFORME').length;
      const moy=ts.length?+(ts.reduce((a,t)=>a+(t.taux||0),0)/ts.length).toFixed(1):0;
      const last=ts.length?ts[0].atDate:'—';
      return ['AFG '+f,ts.length,ok,ts.length-ok,moy?moy+'%':'—',last];
    }),
  ]);
  XLSX.utils.book_append_sheet(wb,ws,'Multi-Filiales');
  XLSX.writeFile(wb,`DBS_Synthese_MultiFiliales_${new Date().toISOString().slice(0,10)}.xlsx`);
  sysLog('OK','Synthèse multi-filiales exportée');
}

// ==================== LOGS ====================
function sysLog(level,msg){
  const now=new Date().toLocaleString('fr-FR');
  APP.logs.push({level,msg,time:now,user:APP.user?.username||'system'});
  if(APP.logs.length>500)APP.logs.shift();
  renderLogs();
}
function renderLogs(){
  const el=document.getElementById('sys-log');if(!el)return;
  const lf=document.getElementById('log-level')?.value||'';
  const ls=document.getElementById('log-search')?.value.toLowerCase()||'';
  const filt=APP.logs.filter(l=>(!lf||l.level===lf)&&(!ls||l.msg.toLowerCase().includes(ls)));
  el.innerHTML=filt.slice(-200).map(l=>`<div class="${l.level==='OK'?'lk':l.level==='ERROR'||l.level==='WARN'?l.level==='ERROR'?'le':'lw':'lo'}">[${l.time}] [${l.level}] [${l.user}] ${l.msg}</div>`).join('');
  el.scrollTop=el.scrollHeight;
}
function clearLogs(){if(APP.user?.role==='admin'&&confirm('Effacer tous les logs ?')){APP.logs=[];renderLogs();}}
function exportLogs(){
  const blob=new Blob([APP.logs.map(l=>`[${l.time}] [${l.level}] [${l.user}] ${l.msg}`).join('\n')],{type:'text/plain'});
  const a=document.createElement('a');a.href=URL.createObjectURL(blob);a.download=`DBS_DataVerify_Logs_${new Date().toISOString().slice(0,10)}.txt`;a.click();
}
